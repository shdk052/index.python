המדריך המקיף ללימוד Python

בואו נלמד יחד את שפת התכנות הפופולרית ביותר. Python היא שפה קלה ללימוד, רבת עוצמה ומושלמת למתחילים.

מבוא ל-Python

Python היא שפת תכנות ברמה גבוהה, קלה ללימוד ורבת שימושים. היא משמשת לפיתוח אפליקציות, בינה מלאכותית, מדע נתונים, אוטומציה ועוד.

קהילה עולמית

Python נהנית מקהילה גדולה ותומכת, עם אינספור ספריות וכלים זמינים

קלה ללימוד

תחביר פשוט וברור שמאפשר למתחילים להתחיל לכתוב קוד במהירות

רב-תכליתית

מתאימה למגוון רחב של יישומים - מפיתוח אפליקציות ותוכנות ועד בינה מלאכותית

התוכנית הראשונה ב-Python

בואו נתחיל עם התוכנית הבסיסית ביותר ב-Python - הדפסת "Hello World": 

print("Hello, World!")

# נוכל גם להדפיס בעברית
print("שלום עולם!")

יסודות השפה

בואו נכיר את המושגים הבסיסיים בפייתון:

משתנים

# הגדרת משתנים בסיסיים בפייתון
name = "יוסי"        # משתנה מסוג מחרוזת (טקסט)
age = 25             # משתנה מסוג מספר שלם
height = 1.75        # משתנה מסוג מספר עשרוני
is_student = True    # משתנה מסוג בוליאני (אמת/שקר)

רשימות

# רשימות מאפשרות לנו לשמור כמה ערכים במשתנה אחד
fruits = ["תפוח", "בננה", "תפוז"]  # רשימת מחרוזות
numbers = [1, 2, 3, 4, 5]           # רשימת מספרים

# הוספת פריט חדש לסוף הרשימה
fruits.append("אבטיח")

מבני בקרה

תנאים

# תנאי if-else בודק האם אדם בגיר או קטין
age = 18
if age >= 18:
    print("בגיר")    # יודפס אם הגיל 18 ומעלה
else:
    print("קטין")    # יודפס אם הגיל מתחת ל-18

לולאות

# לולאת for מריצה קוד מספר פעמים
for i in range(5):    # תרוץ 5 פעמים (0 עד 4)
    print(i)

# לולאת while רצה כל עוד התנאי מתקיים
count = 0
while count < 5:      # תרוץ כל עוד count קטן מ-5
    print(count)
    count += 1        # מגדיל את count ב-1 בכל סיבוב

ספריות בפייתון

ספריות בפייתון הן אוספים של קוד מוכן מראש שמאפשרות לנו להשתמש בפונקציות מתקדמות בקלות. כדי להשתמש בספרייה, אנחנו צריכים לייבא אותה בעזרת הפקודה import: 

# ייבוא ספריות נפוצות
import random  # ספרייה ליצירת מספרים אקראיים
import math    # ספרייה לפעולות מתמטיות

# דוגמאות לשימוש
random_number = random.randint(1, 10)  # מספר אקראי בין 1 ל-10
sqrt_result = math.sqrt(16)            # שורש ריבועי של 16

print(f"מספר אקראי: {random_number}")
print(f"שורש ריבועי של 16: {sqrt_result}")

פונקציות

פונקציות הן דרך נוחה לארגן קוד שחוזר על עצמו. במקום לכתוב את אותו קוד שוב ושוב, אנחנו כותבים אותו פעם אחת בתוך פונקציה:

# פונקציה פשוטה שמקבלת שם ומחזירה ברכה
def say_hello(name):
    return f"שלום {name}! נעים להכיר"

# שימוש בפונקציה
greeting = say_hello("דני")
print(greeting)  # יודפס: שלום דני! נעים להכיר

# פונקציה שמחשבת סכום של שני מספרים
def add_numbers(x, y):
    return x + y

result = add_numbers(5, 3)
print(f"5 + 3 = {result}")  # יודפס: 5 + 3 = 8

ספריית Turtle - ציור גרפי בפייתון

ספריית Turtle היא כלי מצוין ללמידת תכנות באופן ויזואלי. באמצעותה נוכל לצייר צורות וציורים על ידי שליטה ב"צב" וירטואלי שנע על המסך.

צורות בסיסיות

נלמד כיצד לצייר צורות בסיסיות כמו ריבוע ומשולש:

import turtle

# יצירת צב חדש
t = turtle.Turtle()

# ציור ריבוע
for _ in range(4):
    t.forward(100)  # התקדם 100 צעדים
    t.right(90)    # פנה ימינה ב-90 מעלות

# ציור משולש
for _ in range(3):
    t.forward(100)  # התקדם 100 צעדים
    t.right(120)   # פנה ימינה ב-120 מעלות

צבעים ומילוי

נלמד כיצד להוסיף צבעים ולמלא צורות:

import turtle

t = turtle.Turtle()

# שינוי צבע העט והמילוי
t.pencolor("blue")     # צבע הקו
t.fillcolor("yellow")  # צבע המילוי

# ציור מעגל מלא
t.begin_fill()         # התחל מילוי
t.circle(50)           # צייר מעגל ברדיוס 50
t.end_fill()           # סיים מילוי

הגדרות מתקדמות

נלמד על הגדרות נוספות שיעזרו לנו בציור:

import turtle

t = turtle.Turtle()

# הגדרות בסיסיות
t.speed(10)        # מהירות הציור (1-10)
t.pensize(3)       # עובי העט
t.shape("turtle")  # צורת הסמן

# הרמת והורדת העט
t.penup()          # הרם את העט
t.goto(100, 100)   # זוז למיקום חדש
t.pendown()        # הורד את העט

פרויקט לדוגמה

נשלב את כל מה שלמדנו ליצירת כוכב צבעוני:

import turtle

def draw_colorful_star():
    t = turtle.Turtle()
    t.speed(10)
    
    # רשימת צבעים
    colors = ["red", "orange", "yellow", "green", "blue"]
    
    # ציור כוכב צבעוני
    for color in colors:
        t.pencolor(color)
        t.fillcolor(color)
        t.begin_fill()
        
        for _ in range(5):
            t.forward(100)
            t.right(144)
            
        t.end_fill()
        t.right(10)  # סיבוב קל ליצירת אפקט

# הפעלת הפונקציה
draw_colorful_star()
טיפים חשובים:
  • תמיד זכרו לייבא את הספרייה בתחילת הקוד: import turtle
  • השתמשו ב-t.clear() כדי לנקות את המסך
  • השתמשו ב-t.reset() כדי לאפס את מצב הצב
  • השתמשו ב-turtle.done() בסוף התוכנית כדי שהחלון לא ייסגר

ספריית Tkinter - יצירת ממשק משתמש גרפי

Tkinter היא ספריית ברירת המחדל של Python ליצירת ממשק משתמש גרפי (GUI). באמצעותה ניתן ליצור חלונות, כפתורים, תיבות טקסט ועוד אלמנטים גרפיים.

יצירת חלון בסיסי

נלמד כיצד ליצור חלון פשוט עם כותרת:

import tkinter as tk

# יצירת חלון ראשי
window = tk.Tk()
window.title("החלון הראשון שלי")
window.geometry("400x300")  # גודל החלון

# הוספת תווית טקסט
label = tk.Label(window, text="שלום עולם!")
label.pack()

# הפעלת החלון
window.mainloop()

כפתורים ואירועים

נלמד כיצד להוסיף כפתורים ולטפל בלחיצות עליהם:

import tkinter as tk

def button_click():
    label.config(text="לחצת על הכפתור!")

window = tk.Tk()
window.title("כפתורים ואירועים")

# יצירת תווית
label = tk.Label(window, text="לחץ על הכפתור")
label.pack(pady=20)

# יצירת כפתור
button = tk.Button(window, text="לחץ כאן", command=button_click)
button.pack()

window.mainloop()

קלט משתמש

נלמד כיצד לקבל קלט מהמשתמש:

import tkinter as tk

def show_name():
    name = entry.get()
    result_label.config(text=f"שלום {name}!")

window = tk.Tk()
window.title("קלט משתמש")

# תיבת טקסט לקלט
entry = tk.Entry(window)
entry.pack(pady=10)

# כפתור לשליחה
button = tk.Button(window, text="שלח", command=show_name)
button.pack(pady=5)

# תווית להצגת התוצאה
result_label = tk.Label(window, text="")
result_label.pack(pady=10)

window.mainloop()

סידור אלמנטים

נלמד על שיטות שונות לסידור אלמנטים בחלון:

import tkinter as tk

window = tk.Tk()
window.title("סידור אלמנטים")

# שימוש ב-Grid
tk.Label(window, text="שם:").grid(row=0, column=0, padx=5, pady=5)
tk.Entry(window).grid(row=0, column=1, padx=5, pady=5)

tk.Label(window, text="גיל:").grid(row=1, column=0, padx=5, pady=5)
tk.Entry(window).grid(row=1, column=1, padx=5, pady=5)

# שימוש ב-Pack
frame = tk.Frame(window)
frame.grid(row=2, column=0, columnspan=2, pady=10)

tk.Button(frame, text="שמור").pack(side=tk.LEFT, padx=5)
tk.Button(frame, text="בטל").pack(side=tk.LEFT, padx=5)

window.mainloop()
טיפים חשובים:
  • תמיד זכרו להשתמש ב-mainloop() בסוף התוכנית
  • השתמשו ב-pack(), grid() או place() לסידור אלמנטים
  • הקפידו על שמות משתנים ברורים לאלמנטים
  • השתמשו ב-pady ו-padx ליצירת מרווחים
